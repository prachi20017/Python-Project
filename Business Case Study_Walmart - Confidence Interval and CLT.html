male_std = np.std(male_sample, ddof=1)
In [18]:
# Compute 95% confidence intervals
confidence_level = 0.95
z_score = stats.norm.ppf(1 - (1 - confidence_level) / 2)

female_margin_error = z_score * (female_std / np.sqrt(sample_size))
male_margin_error = z_score * (male_std / np.sqrt(sample_size))

female_conf_interval = (female_mean - female_margin_error, female_mean + female_margin_error)
male_conf_interval = (male_mean - male_margin_error, male_mean + male_margin_error)

female_conf_interval, male_conf_interval
